## Code

```javascript
const {
  Worker,
  isMainThread,
  parentPort,
  workerData,
} = require("worker_threads");

let w = new Worker(path, { workerData: null });
w.on("message", (msg) => console.log(msg));
w.on("error", (msg) => console.log(msg));
w.on("exit", (msg) => console.log(msg));
```
